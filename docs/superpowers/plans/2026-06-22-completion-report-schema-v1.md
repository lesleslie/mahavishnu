# Completion Report Schema v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the v1.0 contract that Claude-emitter workers publish to the EventBus after each reasoning iteration and at workflow completion — enabling downstream Phase 1 specs (precommitment-hypothesis-lock, confidence-ceiling-gate) and Phase 2 (three-layer-self-heal) to consume a typed, validated report.

**Architecture:** JSON Schema is canonical at `mahavishnu/core/schemas/completion_report/v1.json`. Pydantic models are generated from the JSON Schema via `datamodel-code-generator`. Workers publish `EventEnvelope(payload=IterationReport|WorkflowReport)` via two new EventBus publisher helpers; a Dhara-backed subscriber auto-persists every validated report. MCP retrieval tools expose iteration history. Migration CLI flags workers that don't emit reports.

**Tech Stack:** Python 3.13, Pydantic v2, `datamodel-code-generator`, `jsonschema` for validation, typer for CLI, FastMCP for MCP tools, pytest with `asyncio_mode = "auto"`, Dhara (existing) for persistence, Prometheus client (existing) for metrics.

## Global Constraints

These apply to every task. Copied verbatim from `pyproject.toml`, `CLAUDE.md`, and the v1 spec.

- **Line length:** 100 chars max.
- **Function args:** ≤ 10 (excludes `self`, `cls`, `*args`, `**kwargs`).
- **Branches:** ≤ 15 per function.
- **Returns:** ≤ 6 per function.
- **Statements:** ≤ 55 per function (practical target 30).
- **Imports:** sorted within each section (stdlib → third-party → first-party with `force-sort-within-sections = true`, `known-first-party = ["mahavishnu"]`).
- **`from __future__ import annotations`** as first non-comment line of every source file (after module docstring).
- **Modern syntax:** `X | None` (not `Optional[X]`), `list[str]` (not `List[str]`), `pathlib.Path` for filesystem paths.
- **Function defaults of `None`:** typed `X | None = None` (mypy `no_implicit_optional = true`).
- **No `assert` in `mahavishnu/**`.** Use the exception hierarchy in `mahavishnu/core/errors.py`.
- **No `Any` in tool inputs or orchestration state.**
- **`except` blocks:** use `logger.exception(...)`, never `logger.error(..., exc_info=True)`.
- **All I/O in the orchestration layer is async.**
- **Use the Oneiric logger** (`oneiric.logging`), not stdlib `logging` or `print()`.
- **Test markers:** `unit`, `integration`, `mcp`. Async tests do not need `@pytest.mark.asyncio` (`asyncio_mode = "auto"`).
- **Coverage target for new modules:** ≥ 95% line coverage on `tests/unit/test_completion_report.py`.
- **Schema strictness:** `additionalProperties: false` in v1; evolution requires explicit schema bump.

```python
class MahavishnuReportValidationError(MahavishnuError):
    """Raised when a worker publishes a report that fails JSON Schema validation.

    Carries the offending payload (sanitized by the publisher) and the list
    of validation error messages for diagnostic context.
    """

    def __init__(self, payload: dict, errors: list[str]) -> None:
        self.payload = payload
        self.errors = errors
        super().__init__(
            f"Completion report validation failed: {'; '.join(errors)}"
        )
```

```python
"""Generate Pydantic models from the completion report JSON Schema.

Run: python scripts/generate_completion_report_model.py

Regenerates mahavishnu/core/completion_report.py from
mahavishnu/core/schemas/completion_report/v1.json.
"""

from __future__ import annotations

from pathlib import Path

from datamodel_code_generator import (
    DataModelType,
    InputFileType,
    generate,
)

REPO_ROOT = Path(__file__).parents[1]
SCHEMA = REPO_ROOT / "mahavishnu" / "core" / "schemas" / "completion_report" / "v1.json"
OUTPUT = REPO_ROOT / "mahavishnu" / "core" / "completion_report.py"


def main() -> None:
    generate(
        input_=SCHEMA,
        input_file_type=InputFileType.JsonSchema,
        output=OUTPUT,
        output_model_type=DataModelType.PydanticV2BaseModel,
        output_model=DataModelType.PydanticV2BaseModel,
        use_schema_description=True,
        field_include_all_keys=True,
        disable_timestamp=True,
        target_python_version=DataModelType.PydanticV2BaseModel,
    )
    print(f"Wrote {OUTPUT}")


if __name__ == "__main__":
    main()
```

```python
"""Pure-function JSON Schema validation for completion reports.

Loads the canonical schema once at import time and exposes validate_*
functions that raise MahavishnuReportValidationError on invalid input.
"""

from __future__ import annotations

import json
from functools import lru_cache
from pathlib import Path

from jsonschema import Draft202012Validator
from jsonschema.exceptions import ValidationError as JSONSchemaValidationError

from mahavishnu.core.errors import MahavishnuReportValidationError

SCHEMA_PATH = (
    Path(__file__).parents[1]
    / "schemas"
    / "completion_report"
    / "v1.json"
)


@lru_cache(maxsize=1)
def _iter_validator() -> Draft202012Validator:
    schema = json.loads(SCHEMA_PATH.read_text())
    return Draft202012Validator(schema["iteration_report"])


@lru_cache(maxsize=1)
def _workflow_validator() -> Draft202012Validator:
    schema = json.loads(SCHEMA_PATH.read_text())
    return Draft202012Validator(schema["workflow_report"])


def _format_errors(errors: list[JSONSchemaValidationError]) -> list[str]:
    return [f"{'/'.join(str(p) for p in e.absolute_path)}: {e.message}" for e in errors]


def validate_iteration_report(report: dict) -> None:
    """Validate an IterationReport against the canonical JSON Schema.

    Raises:
        MahavishnuReportValidationError: if the report does not conform.
    """
    errors = list(_iter_validator().iter_errors(report))
    if errors:
        raise MahavishnuReportValidationError(
            payload=report, errors=_format_errors(errors)
        )


def validate_workflow_report(report: dict) -> None:
    """Validate a WorkflowReport against the canonical JSON Schema.

    Raises:
        MahavishnuReportValidationError: if the report does not conform.
    """
    errors = list(_workflow_validator().iter_errors(report))
    if errors:
        raise MahavishnuReportValidationError(
            payload=report, errors=_format_errors(errors)
        )
```

```python
"""EventBus publishers for completion reports.

Workers call publish_iteration_report() after each Claude turn and
publish_workflow_report() at workflow completion. Reports are validated
against the canonical JSON Schema before being wrapped in an
EventEnvelope and dispatched.
"""

from __future__ import annotations

from oneiric.logging import get_logger

from mahavishnu.core.events.bus import event_bus
from mahavishnu.core.events.envelope import EventEnvelope
from mahavishnu.core.events.report_validation import (
    validate_iteration_report,
    validate_workflow_report,
)

logger = get_logger(__name__)


async def publish_iteration_report(
    report: dict,
    *,
    source: str,
    correlation_id: str,
) -> None:
    """Validate and publish an IterationReport to the EventBus.

    Raises:
        MahavishnuReportValidationError: if the report does not conform
            to the canonical JSON Schema. The caller (typically the
            worker loop) must not advance to the next iteration.
    """
    validate_iteration_report(report)
    envelope = EventEnvelope(
        event_type="workflow.iteration.completed",
        source=source,
        correlation_id=correlation_id,
        payload=report,
        metadata={"report_kind": "iteration"},
    )
    await event_bus.publish(envelope)
    logger.debug(
        "published iteration report",
        extra={
            "workflow_id": report["workflow_id"],
            "iteration_index": report["iteration_index"],
        },
    )


async def publish_workflow_report(
    report: dict,
    *,
    source: str,
    correlation_id: str,
) -> None:
    """Validate and publish a WorkflowReport to the EventBus.

    Raises:
        MahavishnuReportValidationError: if the report does not conform
            to the canonical JSON Schema.
    """
    validate_workflow_report(report)
    envelope = EventEnvelope(
        event_type="workflow.completed",
        source=source,
        correlation_id=correlation_id,
        payload=report,
        metadata={"report_kind": "workflow"},
    )
    await event_bus.publish(envelope)
    logger.debug(
        "published workflow report",
        extra={"workflow_id": report["workflow_id"]},
    )
```

```python
"""Dhara-backed subscriber that persists completion reports.

Subscribes to workflow.iteration.completed and workflow.completed events.
Validates that each payload matches the canonical schema (defense in
depth — the publisher already validates, but the persister trusts
nothing). Persists rows to dhara.iteration_reports and
dhara.workflow_reports. Exposes async query helpers for the MCP layer.
"""

from __future__ import annotations

import json
from typing import Any

from oneiric.logging import get_logger

from mahavishnu.core.events.report_validation import (
    validate_iteration_report,
    validate_workflow_report,
)
from mahavishnu.core.dhara_client import execute, query

logger = get_logger(__name__)

DDL_ITERATION = """
CREATE TABLE IF NOT EXISTS iteration_reports (
    workflow_id TEXT NOT NULL,
    iteration_index INTEGER NOT NULL,
    payload TEXT NOT NULL,
    received_at TEXT NOT NULL DEFAULT (datetime('now')),
    PRIMARY KEY (workflow_id, iteration_index)
)
"""

DDL_WORKFLOW = """
CREATE TABLE IF NOT EXISTS workflow_reports (
    workflow_id TEXT PRIMARY KEY,
    payload TEXT NOT NULL,
    received_at TEXT NOT NULL DEFAULT (datetime('now'))
)
"""


def _ensure_tables() -> None:
    execute(DDL_ITERATION)
    execute(DDL_WORKFLOW)


async def on_iteration_completed(envelope) -> None:
    _ensure_tables()
    validate_iteration_report(envelope.payload)
    payload_json = json.dumps(envelope.payload)
    execute(
        "INSERT INTO iteration_reports (workflow_id, iteration_index, payload) VALUES (?, ?, ?)",
        (envelope.payload["workflow_id"], envelope.payload["iteration_index"], payload_json),
    )
    logger.debug(
        "persisted iteration report",
        extra={
            "workflow_id": envelope.payload["workflow_id"],
            "iteration_index": envelope.payload["iteration_index"],
        },
    )


async def on_workflow_completed(envelope) -> None:
    _ensure_tables()
    validate_workflow_report(envelope.payload)
    payload_json = json.dumps(envelope.payload)
    execute(
        "INSERT OR REPLACE INTO workflow_reports (workflow_id, payload) VALUES (?, ?)",
        (envelope.payload["workflow_id"], payload_json),
    )
    logger.debug(
        "persisted workflow report",
        extra={"workflow_id": envelope.payload["workflow_id"]},
    )


async def get_iteration_history(workflow_id: str) -> list[dict[str, Any]]:
    _ensure_tables()
    rows = query(
        "SELECT payload FROM iteration_reports WHERE workflow_id = ? ORDER BY iteration_index ASC",
        (workflow_id,),
    )
    return [json.loads(row["payload"]) for row in rows]


async def get_workflow_report(workflow_id: str) -> dict[str, Any] | None:
    _ensure_tables()
    rows = query(
        "SELECT payload FROM workflow_reports WHERE workflow_id = ?",
        (workflow_id,),
    )
    if not rows:
        return None
    return json.loads(rows[0]["payload"])


SUBSCRIPTIONS = {
    "workflow.iteration.completed": on_iteration_completed,
    "workflow.completed": on_workflow_completed,
}


def register(bus) -> None:
    """Register the persister handlers with the given EventBus."""
    for event_type, handler in SUBSCRIPTIONS.items():
        bus.subscribe(event_type, handler)
```

```python
"""Migration CLI: audit workers for completion report emission.

Walks a directory (default: current working directory) and flags any
Python module that defines an async function whose name suggests a
Claude worker (run_iteration, run_loop, etc.) but does not import
publish_iteration_report from mahavishnu.core.events.report_publishers.
"""

from __future__ import annotations

import re
from pathlib import Path

import typer

migrate_app = typer.Typer(help="Migration tooling for completion reports")

_WORKER_PATTERN = re.compile(r"async def (run|run_iteration|run_loop|invoke)\b")
_EMITTER_IMPORT = "publish_iteration_report"


def _scan(path: Path) -> list[Path]:
    """Return paths to .py files that look like worker entrypoints."""
    return [
        p
        for p in path.rglob("*.py")
        if _WORKER_PATTERN.search(p.read_text(errors="ignore"))
    ]


def _emits_reports(path: Path) -> bool:
    return _EMITTER_IMPORT in path.read_text(errors="ignore")


@migrate_app.command("check-reports")
def check_reports(
    path: Path = typer.Option(Path("."), "--path", help="Directory to scan"),
) -> None:
    """Exit non-zero if any worker-like module does not emit reports."""
    workers = _scan(path)
    noncompliant = [w for w in workers if not _emits_reports(w)]
    if noncompliant:
        typer.echo(f"Found {len(noncompliant)} worker(s) not emitting reports:")
        for w in noncompliant:
            typer.echo(f"  - {w.relative_to(path)}")
        raise typer.Exit(code=1)
    typer.echo(f"All {len(workers)} worker(s) compliant.")
```

```python
"""Sample iterative researcher worker.

Demonstrates end-to-end completion report emission: each iteration
publishes an IterationReport; workflow completion publishes a
WorkflowReport. The Claude call is stubbed (returns a deterministic
response); this worker's purpose is to exercise the schema + EventBus
+ persister pipeline, not to call a real LLM.
"""

from __future__ import annotations

import uuid
from datetime import UTC, datetime

from oneiric.logging import get_logger

from mahavishnu.core.events.report_publishers import (
    publish_iteration_report,
    publish_workflow_report,
)

logger = get_logger(__name__)


class IterativeResearcher:
    def __init__(
        self,
        *,
        worker_id: str,
        pool_id: str,
        iteration_budget: int = 5,
    ) -> None:
        self.worker_id = worker_id
        self.pool_id = pool_id
        self.iteration_budget = iteration_budget

    async def _stub_claude_call(self, prompt: str, iteration_index: int) -> dict:
        """Stand-in for a real Claude call. Replace in production."""
        return {
            "status": "COMPLETE" if iteration_index >= 1 else "IN_PROGRESS",
            "confidence": 70 + iteration_index * 10,
            "open_questions": [],
            "unchecked_sources": [],
            "contradiction_register": [],
            "assumption_register": [],
            "adjacent_problems": [],
            "draft_response": f"stub response for iteration {iteration_index}",
        }

    async def run(self, prompt: str) -> dict:
        workflow_id = str(uuid.uuid4())
        started_at = datetime.now(UTC)
        confidence_trajectory: list[int] = []

        for iteration_index in range(self.iteration_budget):
            iter_started = datetime.now(UTC)
            response = await self._stub_claude_call(prompt, iteration_index)
            iter_completed = datetime.now(UTC)
            duration_ms = int(
                (iter_completed - iter_started).total_seconds() * 1000
            )

            iteration_report = {
                "schema_version": "1.0.0",
                "report_kind": "iteration",
                "workflow_id": workflow_id,
                "iteration_index": iteration_index,
                "iteration_budget": self.iteration_budget,
                "started_at": iter_started.isoformat(),
                "completed_at": iter_completed.isoformat(),
                "duration_ms": duration_ms,
                **response,
            }
            await publish_iteration_report(
                iteration_report,
                source=self.worker_id,
                correlation_id=workflow_id,
            )
            confidence_trajectory.append(response["confidence"])

            if response["status"] == "COMPLETE":
                break

        completed_at = datetime.now(UTC)
        workflow_report = {
            "schema_version": "1.0.0",
            "report_kind": "workflow",
            "workflow_id": workflow_id,
            "worker_id": self.worker_id,
            "pool_id": self.pool_id,
            "task_type": "iterative_research",
            "started_at": started_at.isoformat(),
            "completed_at": completed_at.isoformat(),
            "duration_ms": int(
                (completed_at - started_at).total_seconds() * 1000
            ),
            "exit_reason": "complete",
            "iteration_count": len(confidence_trajectory),
            "final_status": "COMPLETE",
            "confidence_trajectory": confidence_trajectory,
        }
        await publish_workflow_report(
            workflow_report,
            source=self.worker_id,
            correlation_id=workflow_id,
        )

        logger.info(
            "iterative researcher complete",
            extra={
                "workflow_id": workflow_id,
                "iteration_count": len(confidence_trajectory),
            },
        )
        return workflow_report
```
